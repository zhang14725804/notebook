### 互斥锁死锁

```golang
/*
	fatal error: all goroutines are asleep - deadlock!
	question,为什么会死锁，不应该是顺序执行么 😅😅😅😅😅😅
*/
var mu sync.Mutex
var chain string

func main() {
	chain = "main"
	A()
	fmt.Println(chain)
}
func A() {
	mu.Lock()
	defer mu.Unlock()
	chain = chain + " --> A"
	B()
}
func B() {
	chain = chain + " --> B"
	C()
}
func C() {
	mu.Lock()
	defer mu.Unlock()
	chain = chain + " --> C"
}
```

### 读写锁死锁

```golang
/*
	fatal error: all goroutines are asleep - deadlock!
	question,为什么会死锁，不应该是顺序执行么 😅😅😅😅😅😅
*/
var mu sync.RWMutex
var count int

func main() {
	go A()
	time.Sleep(2 * time.Second)
	mu.Lock()
	defer mu.Unlock()
	count++
	fmt.Println(count)
}
func A() {
	mu.RLock()
	defer mu.RUnlock()
	B()
}
func B() {
	time.Sleep(5 * time.Second)
	C()
}
func C() {
	mu.RLock()
	defer mu.RUnlock()
}
```

### 双检查实现单例

```golang
/*
	 可以编译，有并发问题，f函数可能会被执行多次
	 question 这他么什么意思
*/
type Once struct {
	m    sync.Mutex
	done uint32
}

func (o *Once) Do(f func()) {
	if o.done == 1 {
		return
	}
	o.m.Lock()
	defer o.m.Unlock()
	if o.done == 0 {
		o.done = 1
		f()
	}
}

func main() {
	once := new(Once)
	once.Do(func() {
		fmt.Println(once.done)
	})
}
```

### Mutex

```golang
/*
	加锁后复制变量，会将锁的状态也复制，所以mu1 其实是已经加锁状态，再加锁会死锁。
	question 😅😅😅
*/
type MyMutex struct {
	count int
	sync.Mutex
}

func main() {
	var mu MyMutex
	mu.Lock()
	var mu2 = mu
	mu.count++
	mu.Unlock()
	mu2.Lock()
	mu2.count++
	mu2.Unlock()
	fmt.Println(mu.count, mu2.count)
}
```

### channel

```golang
/*
	一段时间后总是输出 #goroutines: 2
	question 😅😅😅 他么什么意思
	因为 ch 【未初始化】，写和读都会阻塞，之后被第一个协程重新赋值，导致写的ch 都阻塞。
*/
func main() {
	var ch chan int
	go func() {
		ch = make(chan int, 1)
		ch <- 1
	}()
	go func(ch chan int) {
		time.Sleep(time.Second)
		<-ch
	}(ch)
	c := time.Tick(1 * time.Second)
	for range c {
		fmt.Printf("#goroutines: %d\n", runtime.NumGoroutine())
	}
}
```
