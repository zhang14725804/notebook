### 组合继承

```golang
type People struct{}

func (p *People) ShowA() {
	Println("ShowA")
	p.ShowB()
}
func (p *People) ShowB() {
	Println("ShowB")
}

type Teacher struct {
	People
}

func (p *Teacher) ShowB() {
	Println("Teacher-ShowB")
}
func main() {
	t := Teacher{}
	// ShowA
	// ShowB
	t.ShowA()
}
```

### 值方法和指针方法

```golang
type Integer int

func main() {
	var a Integer = 1
    var b Integer = 2
    // 使用【值】可以调用【值方法】和【指针方法】
	Println(a.Add1(b))
	Println(a.Add2(b))

    var i interface{} = &a
    // 使用【指针】可以调用【值方法】和【指针方法】
	Println(i.(*Integer).Add1(b))
	Println(i.(*Integer).Add2(b))

	var c interface{} = a
	Println(c.(Integer).Add1(b))
    // 🔥🔥🔥 类型断言时，无法使用 【值】调用【指针方法】 
    // cannot call pointer method on i.(Integer)
	Println(c.(Integer).Add2(b))
}
// 值方法
func (a Integer) Add1(b Integer) Integer {
	return a + b
}
// 指针方法(推荐)；（1）暗含this指针； （2）接口是引用类型，实现接口用指针方法
func (a *Integer) Add2(b Integer) Integer {
	return *a + b
}
```

### 【值】和【指针】实现interface接口区别

```golang
type Name interface {
	GetName() string
}
type Person struct {
	Name string
}

// 🔥🔥🔥 用【指针】实现的接口，不能用【值】赋值
func (p *Person) GetName() string {
	return p.Name
}
func main() {
	var name Name
	// cannot use Person literal (type Person) as type Name in assignment
	// Person does not implement Name (GetName method has pointer receiver)
	name = Person{}
	Println(name)
}

// 用【值】实现的接口，可以同时用【指针】和【值】赋值
func (p Person) GetName() string {
	return p.Name
}
func main() {
	var name Name
	name = new(Person)
	name = &Person{}
	name = Person{}
	Println(name)
}
```

### 空接口并等于nil（无论是作为返回值还是参数）

```golang
// 带方法的接口
// 🔥🔥🔥 因为data指向了nil 并不代表interface 是nil， 所以返回值并不为nil
type People interface {
	Show()
}

type Student struct{}

func (stu *Student) Show() {}

//
func livePeople() People {
	var stu *Student
	Println(stu)
	return stu
}
func liveStudent() *Student {
	var stu *Student
	Println(stu)
	return stu
}
func main() {
	if livePeople() == nil {
		Println("AAAA")
	} else {
		Println("BBBB")
	}
	if liveStudent() == nil {
		Println("AAAA")
	} else {
		Println("BBBB")
	}
}
```

```golang
func Foo(x interface{}) {
	if x == nil {
		Println("emptyinterface")
		return
	}
	Println("non-emptyinterface")
}
func main() {
    var x *int = nil
    // non-emptyinterface
	Foo(x)
}
```

### 接口类型断言

```golang
func main() {
	i := GetValue()
	// cannot type switch on non-interface value i (type int)
	switch i.(type) {
	case int:
		Println("int")
	case string:
		Println("string")
	case interface{}:
		Println("interface")
	default:
		Println("unknown")
	}
}
func GetValue() int {
	return 1
}
```
